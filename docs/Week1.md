#### Initials

public class Initials {

        System.out.println("***       **    **              **    **        **");
        System.out.println("****      **     **            **     **        **");
        System.out.println("** **     **      **          **      **        **");
        System.out.println("**  **    **       **        **       **        **");
        System.out.println("**   **   **        **      **        ************");

        System.out.println("**     ** **          **  **          **        **");
        System.out.println("**      ****           ****           **        **");
        System.out.println("**       ***            **            **        **");
    }
}
